P6565 [NOI Online 3 J]最急救助

题意简述：给定一个字符串，求出其中子串 的个数。

那么，直接暴力即可。

int findSub(string str)
{
 	int res=0;
 	for(int i=0;i<str.length();i++){if(str.substr(i,3)=="sos"){res++;}}
    return res;
}

其中 str.substr(i,3) 返回一个字符串，为自下标 开始长度为 的子串。

主函数如下。

int main()
{
    int n,maxNum=0;
    string name,signal;
    vector<string> maxName;
    read(n);
    for(int i=1;i<=n;i++)
    {
        cin>>name>>signal;
        int num=findSub(signal);
        if(num>maxNum)
        {
		    maxNum=num;
		    maxName.clear();
		    maxName.push_back(name);
        }
        else if(num==maxNum){maxName.push_back(name);}
    }
    for(int i=0;i<maxName.size();i++){cout<<maxName[i]<<" ";}
    cout<<endl<<maxNum<<endl;
    return 0;
}

注意如果多个信号子串相同，则要输出全部的人名，为了这一点做了一些特殊处理。

P1596 [USACO10OCT S]Lake Counting

这是下一题的铺垫。

题意简述：给定一个图，求八连通连通块的个数。

先简述一下求连通块的过程吧。

首先输入图，转成二维数组，以样例为例（W 为水，. 为旱）

W........WW.
.WWW.....WWW
....WW...WW.
.........WW.
.........W..
..W......W..
.W.W.....WW.
W.W.W.....W.
.W.W......W.
..W.......W.

广搜的过程如下。

  • 从当前点向八个方向扩展，扩展出八个目标
  • 剪掉目标越界的情况
  • 剪掉目标已经被标记访问过的情况（否则会死循环）
  • 剪掉目标为旱地的情况（不然搜出来整个图就一个连通块）
  • 将目标入队并标记已访问（在接下来的代码中把不同连通块标成了不同的颜色，但在此题中没有必要）

然后一定要注意起始结点要标记访问。

从 开始搜，搜过一轮后就出现了一个连通块。然后找到下一个没被标记访问的水（样例中是 ），从它开始搜，找出第二个连通块。以此类推。

代码如下。

struct Point
{
    int x;
    int y;
};
queue<Point> q;
int movX[9]={0,0,0,-1,1,-1,-1,1,1}; // 这是一个小技巧，用数组表示出了每一次移动的坐标变化，下面就一次循环解决
int movY[9]={0,-1,1,0,0,-1,1,-1,1};
int n,m;
bool water[1501][1501];
int dict[1501][1501];

bool out(Point p){return p.x<1 || p.x>n || p.y<1 || p.y>m;} // 判越界
void bfs(Point start,int color)
{
	dict[start.x][start.y]=color;
    q.push(start);
    while(!q.empty())
    {
        Point now=q.front(); // 当前结点
        q.pop();
        for(int i=1;i<=8;i++)
        {
            Point newp; // 目标结点
            newp.x=now.x+movX[i];
            newp.y=now.y+movY[i];
            if(out(newp)){continue;}
            if(!water[newp.x][newp.y]){continue;}
            if(dict[newp.x][newp.y]!=0){continue;}
            dict[newp.x][newp.y]=color;
            q.push(newp);
		}
    }
}

主函数如下。

int main()
{
 	read(n); read(m);
 	string line;
 	for(int i=1;i<=n;i++)
 	{
        cin>>line;
        for(int j=0;j<line.length();j++)
        {
		    if(line[j]=='W'){water[i][j+1]=true;}
		    else if(line[j]=='.'){water[i][j+1]=false;}
		}
    }
	int sum=0;
    for(int i=1;i<=n;i++)
    {
		for(int j=1;j<=m;j++)
		{
            if(dict[i][j]==0 && water[i][j]){bfs((Point){i,j},++sum);}
		}
    }
    cout<<sum<<endl;
 	return 0;
}

P6566 [NOI Online 3 J]观星

观的我两眼冒金星。

题意简述：给定一个图，将图中星星数量相同的八连通连通块合称为星系，求星系的数量及含星星数量最多的星系。

读完题，然后忘了 怎么写。

于是硬调了一个小时多。

然后突然脑抽觉得二维数组开 会爆，结果用了 map，然后出现了玄学的编译错误。

最后调了半天。

有了上题的铺垫，本题应该很容易了，只需要搜出连通块（在搜连通块的时候顺带统计一下连通块大小）然后暴力统计星系即可。使用桶优化至 。

加了桶的主函数代码如下。

int bucket[100005];

int main()

		{
            if(dict[i][j]==0 && star[i][j])
			{
		        int temp=bfs((Point){i,j},++sum);
		        bucket[temp]++; // 桶
		        maxCnt=max(maxCnt,temp);
			}
		}
    }	
    int maxx=0,cnt=0;
    for(int i=1;i<=maxCnt;i++) // 因为不可能有更大的星座，所以只需要 for 到 maxCnt，在一些数据下起到了优化作用
    {
		if(bucket[i]!=0){cnt++;}
	    maxx=max(maxx,bucket[i]*i); // bucket[i]*i 即星系内星星个数
	}
    cout<<cnt<<" "<<maxx<<endl;
 	return 0;
}
